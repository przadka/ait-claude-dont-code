from claude_code_sdk import ClaudeSDKClient, ClaudeCodeOptions

  async def research_agent():
      options = ClaudeCodeOptions(
          system_prompt="Research assistant with access to qualitative data.",
          mcp_servers={"qualbot": {"command": "qualbot-mcp"}},
          allowed_tools=["mcp__qualbot__query", "mcp__qualbot__status"]
      )

      async with ClaudeSDKClient(options=options) as client:
          while True:
              query = input("Research question: ")
              await client.query(query)

              async for response in client.receive_response():
                  print(response.content)

  asyncio.run(research_agent())
